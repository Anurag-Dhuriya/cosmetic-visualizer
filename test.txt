<script type="text/babel">
const { useState } = React;

function App() {
  const [previewImage, setPreviewImage] = useState(null);
  const [activeCategory, setActiveCategory] = useState("Face");

  const categories = ["Face", "Lips", "Nose", "Eyebrows"];

  const handleUpload = (e) => {
    const file = e.target.files[0];
    if (file) {
      setPreviewImage(URL.createObjectURL(file));
    }
  };

  return (
    <div className="h-screen w-screen flex flex-col overflow-hidden bg-app-bg">

      {/* HEADER */}
      <div className="flex items-center justify-between px-4 py-3 border-b border-border-def bg-card-bg shrink-0">
        <div className="flex items-center gap-3">
          <img src="/orion.png" className="h-8 w-8 object-contain" />
          <h1 className="text-lg font-semibold text-text-primary">Orion</h1>
        </div>

        <div className="flex gap-2">
          <button className="px-4 py-2 bg-deep-blue hover:bg-royal-blue text-white rounded">
            Save
          </button>
          <button className="px-4 py-2 bg-[#1F2937] hover:bg-[#374151] text-text-second rounded">
            History
          </button>
        </div>
      </div>

      {/* MAIN LAYOUT */}
      <div className="flex flex-1 overflow-hidden">

        {/* SIDEBAR */}
        <div className="w-64 h-full bg-sidebar-bg border-r border-border-def flex flex-col">

          <div className="p-4 space-y-2">
            {categories.map(cat => (
              <div
                key={cat}
                onClick={() => setActiveCategory(cat)}
                className={`
                  px-3 py-2 rounded cursor-pointer
                  ${activeCategory === cat 
                    ? "bg-royal-blue text-white"
                    : "text-text-muted hover:bg-[#1e293b]"
                  }
                `}
              >
                {cat}
              </div>
            ))}
          </div>

          <div className="flex-1 overflow-y-auto p-4 text-text-muted text-sm">
            History (scrollable)
          </div>
        </div>

        {/* MAIN CANVAS */}
        <div className="flex-1 min-w-0 bg-canvas-bg flex items-center justify-center">

          <div className="w-full h-full flex flex-col items-center justify-center p-4 gap-4">

            <label className="px-4 py-2 bg-royal-blue text-white rounded cursor-pointer">
              Upload Photo
              <input type="file" onChange={handleUpload} hidden />
            </label>

            {previewImage ? (
              <img 
                src={previewImage}
                className="max-h-full max-w-full object-contain"
              />
            ) : (
              <div className="text-text-muted">
                Upload a photo to get started
              </div>
            )}
          </div>
        </div>

        {/* RIGHT PANEL */}
        <div className="w-80 h-full bg-card-bg border-l border-border-def overflow-y-auto p-4 space-y-4">

          <div>
            <p className="text-text-muted text-sm">Selected Category</p>
            <p className="text-text-primary">{activeCategory}</p>
          </div>

          <div>
            <p className="text-text-muted text-sm mb-2">Intensity</p>
            <input type="range" className="w-full" />
          </div>

          <div className="flex gap-2">
            <button className="flex-1 px-4 py-2 bg-deep-blue hover:bg-royal-blue text-white rounded">
              Preview
            </button>
            <button className="flex-1 px-4 py-2 bg-gradient-to-r from-cyan-accent to-royal-blue text-white rounded">
              Apply Final
            </button>
          </div>

        </div>

      </div>
    </div>
  );
}

ReactDOM.createRoot(document.getElementById("root")).render(<App />);
</script>
